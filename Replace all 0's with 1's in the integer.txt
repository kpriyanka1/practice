
//Replace all 0's with 1's in the integer

public class replace {


    public static int replace(int num) {

       if (num==0)
           return 0;

       int digit = num%10;

            if(digit==0)
                digit =9;

           return replace(num/10)*10+digit;

    }

    public static void main(String[] args) {
            int num =100400;

        System.out.println(replace(num));

    }
}
